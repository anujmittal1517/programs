#include <iostream>

using namespace std;


int main()
 {

	// your code goes here

	int i,k,j,a[30],n;

	cin>>n;

	for(i=0;i<n;i++)

	{

	    cin>>a[i];

	}

	for(k=0;k<n;k++)

	{

	    int kmin=k;

	    for(j=k+1;j<n;j++)

	    {

	        if(a[kmin]>a[j])

	        {

	            kmin=j;

	        }

	    }

	    swap(a[k],a[kmin]);

	}

	cout<<"elements are=";

	for(i=0;i<n;i++)

	{

	    cout<<a[i]<<endl;

	}

	return 0;

}
